\section{Triển khai hệ thống lưu trữ phân tán}
 
\subsection{Chuẩn bị phần cứng: }
\begin{itemize}
    \item 1 bộ Switch và dây LAN CAT6E để kết nối các máy tính với nhau
    \item 3 máy tính cài đặt hệ điều hành Microsoft Window 11 Pro Edition kết nối với nhau thông qua Switch
    \begin{itemize}
        \item 1 Name Node (Master): Core i5 9500 (6 Core), GeForce GT 730, 16GB RAM, 1TB SSD
        \item 1 Data Node (Slave): Raspberry Pi 4 Model B, Cortex-A72 (4 Core), 8GB RAM, 128GB SD Card 
    \end{itemize}
\end{itemize}
\subsection{Cài đặt địa chỉ IP, các thư viện cần thiết và phần mềm hệ thống cơ sở dữ liệu phân tán trên các máy tính trong mạng}
\begin{itemize}
    \item Cài đặt trên tất cả các máy Node.js 18.20.4 và npm 10.7.0
    \item Cài đặt địa chỉ IPv4 cho các máy:
    \begin{itemize}
        \item Name Node: 192.168.1.88
        \item Data Node 1: 192.168.1.8
        \item Data Node 2: 192.168.1.2
    \end{itemize}
    \item Cài đặt bộ mã nguồn phần mềm hệ thống cơ sở dữ liệu phân tán trên cả tất cả các máy tính.
    \item Tạo một thư mục có tên store trên 2 Data Node để lưu trữ dữ liệu.   
\end{itemize}

 Khi client nhận diện chuyển động và tự động cắt video 20s và gửi lên cho máy chủ nhận diện hành động.Hành động được dự đoán là typing với độ tin cậy là 31.56\%.
